<script>

// BASE PRODUKTER
let products = JSON.parse(localStorage.getItem("products")) || [
"Koreansk oljerens",
"Mild vannrens",
"BOJ Ginseng Essence",
"d’Alba Mist",
"Vitamin C",
"Laneige Serum",
"Paula’s Choice Cream",
"Byoma Serum",
"Round Lab Cream",
"Kérastase Oil",
"Gisou Oil",
"Delícia Body Butter"
];

// RUTINE LOGIKK (forenklet smart system)
function getRoutine(week, day, time) {

let routine = [];

// MORGEN
if(time === "Morgen") {
    routine = [
        "Rens",
        "Toner",
        "Mist",
        "Vitamin C",
        "Serum",
        "Øyekrem",
        "Fuktighetskrem",
        "Lepper"
    ];
}

// KVELD
if(time === "Kveld") {

    routine.push("Dusj");
    routine.push("Barbering");

    // SKRUBB
    if(day === "Torsdag") {
        routine.push("Body scrub");
        routine.push("Body oil");
    } else {
        routine.push("Body cream");
    }

    // HÅRVASK
    if(week == 1 && ["Mandag","Onsdag","Fredag","Søndag"].includes(day)) {
        routine.push("Hårvask");
    }

    if(week == 2 && ["Tirsdag","Torsdag","Lørdag"].includes(day)) {
        routine.push("Hårvask");
    }

    routine.push("Rens");
    routine.push("Toner");

    if(day === "Torsdag" && week == 1) {
        routine.push("Peel");
    }

    routine.push("Serum");
    routine.push("Øyekrem");
    routine.push("Fuktighetskrem");
    routine.push("Lepper");
}

return routine;
}

// VIS RUTINE
function showRoutine() {
let week = document.getElementById("week").value;
let day = document.getElementById("day").value;
let time = document.getElementById("time").value;

let routine = getRoutine(week, day, time);

document.getElementById("routine").innerHTML =
routine.map(step => "• " + step).join("<br>");
}

// LEGG TIL PRODUKT
function addProduct() {
let input = document.getElementById("newProduct").value;

if(input) {
    products.push(input);
    localStorage.setItem("products", JSON.stringify(products));
    renderProducts();
}
}

// VIS PRODUKTER
function renderProducts() {
let list = document.getElementById("productList");
list.innerHTML = "";

products.forEach((p, index) => {
    let li = document.createElement("li");
    li.innerHTML = p + 
    ` <button onclick="removeProduct(${index})">X</button>`;
    list.appendChild(li);
});
}

// FJERN PRODUKT
function removeProduct(index) {
products.splice(index, 1);
localStorage.setItem("products", JSON.stringify(products));
renderProducts();
}

renderProducts();
